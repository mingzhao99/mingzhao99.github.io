initial setup for local repository

# from git bash
# setup username
git config --global user.name 'mingzhao99'

#setup user email
git config --global user.email 'mingzhao99@gmail.com'

# check config
git config --list

# initial repository: from git bash mode
$ mkdir test
$ cd test
$ git init
Initialized empty Git repository in C:/Users/ming_/OneDrive/Desktop/git_test/test/.git/

# create a file t.sh under test folder
git status
# add t.sh from working directory to temp
git add t.sh
git status

# move from temp to repository
git commit -m "create t.sh"

# modify file t.sh, and run
git add t.sh
git commit -m "update t.sh"

# delete t.sh, and run
rm t.sh
git commit -m "delete t.sh"
